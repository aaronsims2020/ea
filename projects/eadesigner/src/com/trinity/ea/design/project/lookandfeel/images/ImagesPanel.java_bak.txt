    public synchronized void getDataUpdate()
    {
	  try
	  {
		// Splash Image Size
            SpinnerNumberModel splashImageSMWidth = (SpinnerNumberModel)spinSplashImagePixelWidth.getModel();
            SpinnerNumberModel splashImageSMHeight = (SpinnerNumberModel)spinSplashImagePixelHeight.getModel();
            ProjectManager.putTempNoFileWrite("splashImageSize", String.valueOf(splashImageSMWidth.getNumber().intValue()) + "," + String.valueOf(splashImageSMHeight.getNumber().intValue()));            
		// Override Splash Image Size
		if(SplashImagePanelOverideDefaultPanelSizeEnabled.isSelected()==true)
		{
            	SpinnerNumberModel overrideSplashImageSMWidth = (SpinnerNumberModel)spinOverrideDefaultPixelWidth.getModel();
            	SpinnerNumberModel overrideSplashImageSMHeight = (SpinnerNumberModel)spinOverrideDefaultPixelWidtHeight.getModel();
            	ProjectManager.putTempNoFileWrite("splashImgOverideDefaultPanelSize", "true");            
            	ProjectManager.putTempNoFileWrite("splashImgOveridePanelWidth", String.valueOf(overrideSplashImageSMWidth.getNumber().intValue()));            
            	ProjectManager.putTempNoFileWrite("splashImgOveridePanelHeight", String.valueOf(overrideSplashImageSMHeight.getNumber().intValue()));            
		}
		else
		{
            	ProjectManager.putTempNoFileWrite("splashImgOverideDefaultPanelSize", "false");
            	ProjectManager.putTempNoFileWrite("splashImgOveridePanelWidth", "");            
            	ProjectManager.putTempNoFileWrite("splashImgOveridePanelHeight", ""); 
		}
		// Override Vertical Button Bar Size
		if(verticalButtonBarOverideDefaultPanelSizeEnabled.isSelected()==true)
		{
            	SpinnerNumberModel overrideSplashImageSMWidth2 = (SpinnerNumberModel)spinOverrideDefaultPixelWidth1.getModel();
            	SpinnerNumberModel overrideSplashImageSMHeight2 = (SpinnerNumberModel)spinOverrideDefaultPixelWidtHeight1.getModel();
            	ProjectManager.putTempNoFileWrite("btnBarImgOverideDefaultPanelSize", "true");            
            	ProjectManager.putTempNoFileWrite("btnBarImgOveridePanelWidth", String.valueOf(overrideSplashImageSMWidth2.getNumber().intValue()));            
            	ProjectManager.putTempNoFileWrite("btnBarImgOveridePanelHeight", String.valueOf(overrideSplashImageSMHeight2.getNumber().intValue()));            
		}
		else
		{
            	ProjectManager.putTempNoFileWrite("btnBarImgOverideDefaultPanelSize", "false");
            	ProjectManager.putTempNoFileWrite("btnBarImgOveridePanelWidth", "");            
            	ProjectManager.putTempNoFileWrite("btnBarImgOveridePanelHeight", ""); 
		}
		// Override Progress Bar Image Panel Size
		if(progressBarImageOverideDefaultPanelSizeEnabled.isSelected()==true)
		{
            	SpinnerNumberModel overrideSplashImageSMWidth3 = (SpinnerNumberModel)spinProgressBarImagePanelOverrideDefaultPixelWidth.getModel();
            	SpinnerNumberModel overrideSplashImageSMHeight3 = (SpinnerNumberModel)spinProgressBarImagePanelOverrideDefaultPixelHeight.getModel();
            	ProjectManager.putTempNoFileWrite("progressPanelImgOverideDefaultPanelSize", "true");            
            	ProjectManager.putTempNoFileWrite("progressPanelImgOveridePanelWidth", String.valueOf(overrideSplashImageSMWidth3.getNumber().intValue()));            
            	ProjectManager.putTempNoFileWrite("progressPanelImgOveridePanelHeight", String.valueOf(overrideSplashImageSMHeight3.getNumber().intValue()));            
		}
		else
		{
            	ProjectManager.putTempNoFileWrite("progressPanelImgOverideDefaultPanelSize", "false");
            	ProjectManager.putTempNoFileWrite("progressPanelImgOveridePanelWidth", "");            
            	ProjectManager.putTempNoFileWrite("progressPanelImgOveridePanelHeight", ""); 
		}
		if(cbActionType.getSelectedIndex()==0)
		{
			ProjectManager.putTempNoFileWrite("btnBarImgActionType", "-1"); 
		}
		else if(cbActionType.getSelectedIndex()==1)
		{
			ProjectManager.putTempNoFileWrite("btnBarImgActionType", "0"); 
		}	
		else if(cbActionType.getSelectedIndex()==2)
		{
			ProjectManager.putTempNoFileWrite("btnBarImgActionType", "1"); 
		}	
		ProjectManager.putTempNoFileWrite("btnBarImgAction", tfVerticalButtonBarImage1.getText()); 
		if(cbProgressBarImageActionType.getSelectedIndex()==0)
		{
			ProjectManager.putTempNoFileWrite("progressPanelImgActionType", "-1"); 
		}
		else if(cbProgressBarImageActionType.getSelectedIndex()==1)
		{
			ProjectManager.putTempNoFileWrite("progressPanelImgActionType", "0"); 
		}	
		else if(cbProgressBarImageActionType.getSelectedIndex()==2)
		{
			ProjectManager.putTempNoFileWrite("progressPanelImgActionType", "1"); 
		}	
		ProjectManager.putTempNoFileWrite("progressPanelImgAction", tfProgressBarImageAction.getText()); 


            ProjectManager.saveTempNow(); 
        }
        catch(Exception e)
        {
            e.printStackTrace();
        }
    }

    private void ProgressBarImagePanelColorChooserButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_ProgressBarImagePanelColorChooserButtonActionPerformed
 	// Bring up a color chooser 
        if(ProjectManager.get("progressPanelImgBGColor")!=null)
        {
            if(ProjectManager.get("progressPanelImgBGColor").equalsIgnoreCase("")==false)
            {
                Object[] tmpArray = getStringArraysFromString(ProjectManager.get("progressPanelImgBGColor"));
                Color c = JColorChooser.showDialog(this,"Color Chooser",new java.awt.Color(new Integer((String)tmpArray[0]).intValue(), new Integer((String)tmpArray[1]).intValue(),new Integer((String)tmpArray[2]).intValue()));

                ProgressBarImagePanelColorChooserButton.setIcon(new ColorSwatch(c));
            }
            else
            {
                Color c = JColorChooser.showDialog(this,"Color Chooser",new java.awt.Color(255,255,255)); 
                ProjectManager.put("progressPanelImgBGColor", String.valueOf(c.getRed()) + "," + String.valueOf(c.getGreen()) + "," + String.valueOf(c.getBlue()));                
                ProgressBarImagePanelColorChooserButton.setIcon(new ColorSwatch(c));
            }
        }
        else
        {
                Color c = JColorChooser.showDialog(this,"Color Chooser",new java.awt.Color(255,255,255)); 
                ProjectManager.put("progressPanelImgBGColor", String.valueOf(c.getRed()) + "," + String.valueOf(c.getGreen()) + "," + String.valueOf(c.getBlue()));                
                ProgressBarImagePanelColorChooserButton.setIcon(new ColorSwatch(c));
        }
    }//GEN-LAST:event_ProgressBarImagePanelColorChooserButtonActionPerformed

    private static Object[] getStringArraysFromString(String textArrayString)
    {
        ArrayList aryList = new ArrayList();
        String tempString = "";
        
        while(textArrayString.indexOf(",")!=-1)
        {
            tempString = textArrayString.substring(0,textArrayString.indexOf(","));
            textArrayString = textArrayString.substring(textArrayString.indexOf(",") + 1);
            aryList.add(tempString);
        }
        aryList.add(textArrayString);
        return aryList.toArray();
    }
